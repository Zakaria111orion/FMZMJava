51      /**
52  	 * CrystalBall code to controller order.
53  	 *
54  	 * @param keyCode
55  	 *          the key code
56  	 * @return the controller order
57  	 */
58  	protected static ControllerOrder keyCodeToControllerOrder(final int keyCode) {
59  		switch (keyCode) {
60  			case KeyEvent.VK_NUMPAD1:
61  				return ControllerOrder.MAP1;
62  			case KeyEvent.VK_NUMPAD2:
63  				return ControllerOrder.MAP2;
64  			case KeyEvent.VK_NUMPAD3:
65  				return ControllerOrder.MAP3;
66  			case KeyEvent.VK_NUMPAD4:
67  				return ControllerOrder.MAP4;
68              case KeyEvent.VK_NUMPAD5:
69                  return ControllerOrder.MAP5;
70  			case KeyEvent.VK_NUMPAD6:
71  				return ControllerOrder.MAP6;
72  			case KeyEvent.VK_NUMPAD7:
73  				return ControllerOrder.MAP7;
74  			case KeyEvent.VK_NUMPAD8:
75  				return ControllerOrder.MAP8;
76  			case KeyEvent.VK_NUMPAD9:
77  				return ControllerOrder.MAP9;
78  			case KeyEvent.VK_NUMPAD0:
79  				return ControllerOrder.TEST;
80              case KeyEvent.VK_M:
81                  return ControllerOrder.MENU;
82              case KeyEvent.VK_W:
83                  return ControllerOrder.WORKSHOP;
84  			case KeyEvent.VK_LEFT:
85  				return ControllerOrder.MOVELEFT;
86  			case KeyEvent.VK_RIGHT:
87  				return ControllerOrder.MOVERIGHT;
88  			case KeyEvent.VK_DOWN:
89  				return ControllerOrder.MOVEDOWN;
90  			case KeyEvent.VK_UP:
91  				return ControllerOrder.MOVEUP;
92              case KeyEvent.VK_SPACE:
93                  return ControllerOrder.FIRE;
94  			default:
95  				return null;
96  		}
97  	}
